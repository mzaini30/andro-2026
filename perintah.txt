Pada ./andro.bat:

REM Create andro.yml
echo Creating andro.yml...
(
echo - title: ""
echo - version: "1"
echo - package: "com.mzaini30."
echo - icon: ""
echo - web: "build"
echo - ads: ""
) > "%CURRENT_DIR%\andro.yml"

- title: ""

  - id: ""
  - banner: ""
  - open: ""
